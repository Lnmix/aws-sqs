 21 /**
 22  * Convience function for making a request to the SQS service.
 23  * @param  {object}   aws    AWS service from aws-lib.
 24  * @param  {string}   action The AWS action to invoke on the service.
 25  * @param  {object}   params The parameters to pass the action.
 26  * @param  {Function} fn Callback function from the service call, will include
 27  * an error (or null) and the result of the service call.
 28  * @private
 29  */
 30 var makeCall = function(aws, action, params, fn) {
 31   aws.call(action, params, function(err, result) {
 32     if(!err && result && result.Error && result.Error.Message) {
 33       err = result.Error.Message;
 34     }
 35     fn(err, result);
 36   });
 37 };

 39 /**
 40  * Simple Queue Service, used to communicate and run commands against Amazon's
 41  * SQS.
 42  * @example
 43  *   var sqs = new SQS('awsId', 'awsSecret');
 44  * @param {string} accessKeyId key id
 45  * @param {string} secretAccessKey secret access key
 46  * @param {object} [opts] optional parameters
 47  * @config {string} [opts.region] the aws region, must be one of these:
 48  *   us-east-1, us-west-1, us-west-2, eu-west-1, ap-southeast-1, ap-northeast-1
 49  * @class
 50  */
 51 var SQS = function(accessKeyId, secretAccessKey, opts) {
 52 
 53   // optional parameters
 54   opts = opts || {};
 55   this.accessKeyId = accessKeyId;
 56   this.secretAccessKey = secretAccessKey;
 57   this.region = opts.region || 'us-east-1';
 58   this.host = 'sqs.' + this.region + '.amazonaws.com';
 59 };

 61 /**
 62  * Create a SQS queue
 63  * @param  {string} name The name of the queue
 64  *   Constraints: Maximum 80 characters; alphanumeric characters, hyphens (-),
 65  *   and underscores (_) are allowed.
 66  * @param  {object} [attrs] Attributes for the queue.
 67  * @config {number} [attrs.VisibilityTimeout] The length of time (in seconds) that a
 68  * message received from a queue will be invisible to other receiving components
 69  * when they ask to receive messages. For more information about
 70  * VisibilityTimeout, see Visibility Timeout in the Amazon SQS Developer Guide.
 71  * An integer from 0 to 43200 (12 hours). The default for this attribute is 30.
 72  * @config {string} [attrs.Policy] The formal description of the permissions for a resource.
 73  * For more information about Policy, see Basic Policy Structure in the Amazon
 74  * SQS Developer Guide. A valid form-url-encoded policy. For more information
 75  * about policy structure, see Basic Policy Structure in the Amazon SQS
 76  * Developer Guide. For more information about form-url-encoding,
 77  * see http://www.w3.org/MarkUp/html-spec/html-spec_8.html#SEC8.2.1.
 78  * @config {number} [attrs.MaximumMessageSize] The limit of how many bytes a message can
 79  * contain before Amazon SQS rejects it. An integer from 1024 bytes (1 KiB) up
 80  * to 65536 bytes (64 KiB). The default for this attribute is 65536 (64 KiB).
 81  * @config {number} [attrs.MessageRetentionPeriod] The number of seconds Amazon SQS retains a
 82  * message. Integer representing seconds, from 60 (1 minute) to 1209600
 83  * (14 days). The default for this attribute is 345600 (4 days).
 84  * @config {number} [attrs.DelaySeconds] The time in seconds that the delivery of all messages
 85  * in the queue will be delayed. An integer from 0 to 900 (15 minutes). The
 86  * default for this attribute is 0 (zero).
 87  * @param  {Function} fn Callback function for when queue is created, Callback
 88  * function takes two parameters error and result, result will be full name
 89  * for the queue.
 90  * @link SQS.prototype.deleteQueue
 91  */
 92 SQS.prototype.createQueue = function(name, attrs, fn) {
 93   if(!fn) {
 94     fn = attrs;
 95     attrs = null;
 96   }
 97 
 98   // Check name to make sure meets constraints
 99   if(name.length > 80 || name.search(" ") != -1) {
100     fn('Name is constrained to maximum 80 characters; ' +
101        'alphanumeric characters, hyphens (-), and underscores (_).', null);
102     return;
103   }
104 
105   var params = {
106     QueueName : name
107   };
108 
109   if(attrs) {
110     var i = 1;
111     for(var prop in attrs) {
112       params["Attribute." + i + ".Name"] = prop;
113       params["Attribute." + i + ".Value"] = attrs[prop];
114       i++;
115     }
116   }
117 
118   var host = this.host;
119   var aws = awsjs.createSQSClient(
120     this.accessKeyId,
121     this.secretAccessKey,
122     {
123       host : this.host,
124       secure: true,
125       version: '2011-10-01'
126   });
127 
128   makeCall(aws, 'CreateQueue', params, function(err, result) {
129     var url = '';
130 
131     if(result && result.CreateQueueResult) {
132       url = result.CreateQueueResult.QueueUrl;
133     }
134 
135     var name = url.replace('https://' + host, '');
136 
137     fn(err, name);
138   });
139 };

288 /**
289  * Sends a batch of messages to a queue.
290  * @param  {string} queue The full name of the queue. Should be something like
291  * /12231231231/nameOfQueue.
292  * @param  {Array} messages An array of objects, each object should have
293  * message property and optional delay property.
294  * @param  {Function} fn The callback for when the message is add to queue,
295  * Callback function should take two parameters, error and result which is
296  * an array of message details.
297  * @link SQS.prototype.sendMessage
298  */
299 SQS.prototype.sendMessageBatch = function(queue, messages, fn) {
300   var aws = awsjs.createSQSClient(
301     this.accessKeyId,
302     this.secretAccessKey,
303     {
304       host : this.host,
305       path : queue,
306       secure: true
307   });
308 
309   var params = {};
310 
311   for (var i = 0; i < messages.length; i++) {
312     var index = i + 1;
313     params['SendMessageBatchRequestEntry.' + index + '.Id'] = i;
314     params['SendMessageBatchRequestEntry.' + index + '.MessageBody'] = messages[i].message;
315     if(messages[i].delay) {
316       params['SendMessageBatchRequestEntry.' + index + '.DelaySeconds'] = messages[i].delay;
317     }
318   }
319 
320   makeCall(aws, 'SendMessageBatch', params, function(err, result) {
321     if(err) {
322       fn(err, null);
323     } else {
324       var ret = null;
325       if(result.SendMessageBatchResult &&
326          result.SendMessageBatchResult.SendMessageBatchResultEntry)
327       {
328         ret = result.SendMessageBatchResult.SendMessageBatchResultEntry;
329       }
330       fn(err, ret);
331     }
332   });
333 };

335 /**
336  * Retrieves any messages from a queue if any are available. If none are
337  * available then an empty array is returned as the result.
338  * @param  {string} queue The full name of the queue to check for messages,
339  * should look something like /12312301239/nameOfQueue.
340  * @param  {object} [opts] Options for retreiving messages.
341  * @config {number} [opts.MaxNumberOfMessages] The max number of messages to pull
342  * from queue.
343  * @config {number} [opts.VisibilityTimeout] The length of time in seconds that a
344  * message is not visible for receive requests after returned from this one.
345  * @param  {Function} fn The callback for when the message(s) are pulled from
346  * queue, Callback function should take two parameters, error and result which
347  * is an array of messages.
348  * @link SQS.prototype.sendMessage
349  */
350 SQS.prototype.receiveMessage = function(queue, opts, fn) {
351   if(!fn) {
352     fn = opts;
353     opts = null;
354   }
355 
356   var aws = awsjs.createSQSClient(
357     this.accessKeyId,
358     this.secretAccessKey,
359     {
360       host : this.host,
361       path : queue,
362       secure: true
363   });
364 
365   var params = {};
366 
367   if(opts && opts.MaxNumberOfMessages) {
368     params.MaxNumberOfMessages = opts.MaxNumberOfMessages;
369   }
370 
371   if(opts && opts.VisibilityTimeout) {
372     params.VisibilityTimeout = opts.VisibilityTimeout;
373   }
374 
375   if(opts && opts.attrs) {
376     for (var i = 1; i <= opts.attrs.length; i++) {
377       params["AttributeName." + i] = opts.attrs[i-1];
378     }
379   }
380 
381   makeCall(aws, 'ReceiveMessage', params, function(err, result) {
382     if(err) {
383       fn(err, null);
384     } else {
385       var messages = null;
386       if(result.ReceiveMessageResult && result.ReceiveMessageResult.Message) {
387         if(result.ReceiveMessageResult.Message.length) {
388           messages = result.ReceiveMessageResult.Message;
389         } else {
390           messages = [result.ReceiveMessageResult.Message];
391         }
392       }
393       if(messages !== null) {
394         for(var i = 0; i < messages.length; i++) {
395           var message = messages[i];
396           if(message.Attribute && message.Attribute.length) {
397             for(var j = 0; j < message.Attribute.length; j++) {
398               message[message.Attribute[j].Name] = message.Attribute[j].Value;
399             }
400           } else {
401             if(message.Attribute) {
402               message[message.Attribute.Name] = message.Attribute.Value;
403             }
404           }
405         }
406       }
407       fn(err, messages);
408     }
409   });
410 };
